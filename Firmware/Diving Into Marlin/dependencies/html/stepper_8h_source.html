281  static FORCE_INLINE unsigned short calc_timer(unsigned short step_rate) {
282  unsigned short timer;
283 
284  NOMORE(step_rate, MAX_STEP_FREQUENCY);
285 
286  if (step_rate > 20000) { // If steprate > 20kHz >> step 4 times
287  step_rate >>= 2;
288  step_loops = 4;
289  }
290  else if (step_rate > 10000) { // If steprate > 10kHz >> step 2 times
291  step_rate >>= 1;
292  step_loops = 2;
293  }
294  else {
295  step_loops = 1;
296  }
297 
298  NOLESS(step_rate, F_CPU / 500000);
299  step_rate -= F_CPU / 500000; // Correct for minimal speed
300  if (step_rate >= (8 * 256)) { // higher step rate
301  unsigned short table_address = (unsigned short)&speed_lookuptable_fast[(unsigned char)(step_rate >> 8)][0];
302  unsigned char tmp_step_rate = (step_rate & 0x00ff);
303  unsigned short gain = (unsigned short)pgm_read_word_near(table_address + 2);
304  MultiU16X8toH16(timer, tmp_step_rate, gain);
305  timer = (unsigned short)pgm_read_word_near(table_address) - timer;
306  }
307  else { // lower step rates
308  unsigned short table_address = (unsigned short)&speed_lookuptable_slow[0][0];
309  table_address += ((step_rate) >> 1) & 0xfffc;
310  timer = (unsigned short)pgm_read_word_near(table_address);
311  timer -= (((unsigned short)pgm_read_word_near(table_address + 2) * (unsigned char)(step_rate & 0x0007)) >> 3);
312  }
313  if (timer < 100) { // (20kHz - this should never happen)
314  timer = 100;
315  MYSERIAL.print(MSG_STEPPER_TOO_HIGH);
316  MYSERIAL.println(step_rate);
317  }
318  return timer;
319  }
